          speechSynthesis.speak(utterance);
          setTimeout(() => speechSynthesis.cancel(), 2000);
          removeVoiceListeners();
        }
      }

      function speak(text) {
        if (!voiceEnabled && !text.includes("إيقاف")) return;

        speechSynthesis.cancel();
        const utterance = new SpeechSynthesisUtterance(text);
        utterance.lang = "ar-SA";
        utterance.rate = 0.9;
        utterance.pitch = 1;
        utterance.volume = 1;

        utterance.onstart = function () {
          console.log("بدأ القراءة:", text);
        };

        utterance.onerror = function (event) {
          console.error("خطأ في القراءة:", event);
        };

        speechSynthesis.speak(utterance);
      }

      function addVoiceListeners() {
        const cards = document.querySelectorAll(".ipcpo-card");
        cards.forEach((card) => {
          card.addEventListener("mouseenter", handleCardHover);
        });

        const buttons = document.querySelectorAll(
          "button:not(.accessibility-btn)"
        );
        buttons.forEach((btn) => {
          btn.addEventListener("mouseenter", handleButtonHover);
        });
      }

      function handleCardHover(e) {
        if (!voiceEnabled) return;
        const title = this.querySelector(
          ".card-header span:last-child"
        ).textContent;
        speak(title);
      }

      function handleButtonHover(e) {
        if (!voiceEnabled) return;
        const text =
          this.textContent ||
          this.getAttribute("aria-label") ||
          this.getAttribute("title");
        if (text && text.trim()) speak(text.trim());
      }

      function removeVoiceListeners() {
        const cards = document.querySelectorAll(".ipcpo-card");
        cards.forEach((card) => {
          card.removeEventListener("mouseenter", handleCardHover);
        });

        const buttons = document.querySelectorAll(
          "button:not(.accessibility-btn)"
        );
        buttons.forEach((btn) => {
          btn.removeEventListener("mouseenter", handleButtonHover);
        });
      }

      function toggleHighContrast() {
        document.body.classList.toggle("high-contrast");
        const btn = event.target.closest(".accessibility-btn");
        btn.classList.toggle("active");
        const isActive = document.body.classList.contains("high-contrast");
        const message = isActive
          ? "تم تفعيل وضع التباين العالي"
          : "تم إيقاف وضع التباين العالي";

        if (voiceEnabled) {
          speak(message);
        } else {
          const utterance = new SpeechSynthesisUtterance(message);
          utterance.lang = "ar-SA";
          speechSynthesis.speak(utterance);
        }
      }
